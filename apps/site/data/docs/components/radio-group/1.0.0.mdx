```tsx

  RadioGroup,
  RadioGroupIndicator,
  RadioGroupItem,
} from 'tamagui' } from 'tamagui'

  <RadioGroup>
    <RadioGroupItem value={} size="$4">
      <RadioGroupIndicator size={} />
    </RadioGroupItem>

)
```
